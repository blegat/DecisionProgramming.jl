DecisionProgramming.utility_distribution — Function

The probability distribution of utilities.

using Plots
x, y = distribution(z, diagram, params)
y2 = cumsum(y)
p = plot(x, y,
    linestyle=:dash,
    markershape=:circle,
    ylims=(0, 1.1),
    label="Distribution",
    legend=:topleft)
plot!(p, x, y2,
    linestyle=:dash,
    markershape=:circle,
    label="Cumulative distribution")
savefig(p, "distribution.svg")
